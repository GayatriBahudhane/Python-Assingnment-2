# Program 1: Sum of numbers from 1 to 50
num = 51
total = 0

for i in range(num):
    total += i

print("The sum of numbers from 1 to 50 is:", total)

# Program 2: Even or Odd Checker
num2 = int(input("Enter a number: "))

if num2 % 2 == 0:
    print(f"{num2} is an even number")
else:
    print(f"{num2} is an odd number")
